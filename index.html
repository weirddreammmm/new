        function speakResult(text) {

            const utterance = new SpeechSynthesisUtterance(text); // 텍스트를 음성으로 변환

            utterance.lang = 'ko-KR'; // 한국어 설정

            

            // 괄호 안의 내용은 속삭이듯 읽기

            const whisperRegex = /\((.*?)\)/g; // 괄호 내용 정규 표현식

            const splitText = text.split(whisperRegex); // 괄호를 기준으로 텍스트 분리


            splitText.forEach(part => {

                const whisperUtterance = new SpeechSynthesisUtterance(part.trim());

                whisperUtterance.lang = 'ko-KR';

                if (part.startsWith('(') && part.endsWith(')')) {

                    whisperUtterance.volume = 0.2; // 볼륨을 낮춰 속삭이는 효과 추가

                    whisperUtterance.rate = 0.7; // 속도 느리게 설정

                }

                window.speechSynthesis.speak(whisperUtterance); // 음성 출력

            });

        }
